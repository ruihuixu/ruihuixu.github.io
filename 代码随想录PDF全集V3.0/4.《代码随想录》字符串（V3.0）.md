class Solution {


public:

void reverse(string& s, int start, int end){ //翻转，区间写法：左闭右闭 []


for (int i = start, j = end; i < j; i++, j--) {


swap(s[i], s[j]);


}


}


void removeExtraSpaces(string& s) {//去除所有空格并在相邻单词之间添加空格, 快慢指针。

int slow = 0;  //整体思想参考https://programmercarl.com/0027.移除元素.html


for (int i = 0; i < s.size(); ++i) { //

if (s[i] != ' ') { //遇到⾮空格就处理，即删除所有空格。

if (slow != 0) s[slow++] = ' '; //⼿动控制空格，给单词之间添加空格。slow != 0

说明不是第⼀个单词，需要在单词前添加空格。

while (i < s.size() && s[i] != ' ') { //补上该单词，遇到空格说明单词结束。


s[slow++] = s[i++];


}


}


}

s.resize(slow); //slow的⼤⼩即为去除多余空格后的⼤⼩。


}


string reverseWords(string s) {

removeExtraSpaces(s); //去除多余空格，保证单词之间之只有⼀个空格，且字符串⾸尾没空格。


reverse(s, 0, s.size() - 1);

int start = 0; //removeExtraSpaces后保证第⼀个单词的开始下标⼀定是0。


for (int i = 0; i <= s.size(); ++i) {

if (i == s.size() || s[i] == ' ') { //到达空格或者串尾，说明⼀个单词结束。进⾏翻

转。

reverse(s, start, i          - 1); //翻转，注意是左闭右闭 []的翻转。

start = i + 1; //更新下⼀个单词的开始下标start


}


}


return s;


}


};

### **最⻓公共前后缀**


⽂章中字符串的 **前缀是指不包含最后⼀个字符的所有以第⼀个字符开头的连续⼦串** 。


**后缀是指不包含第⼀个字符的所有以最后⼀个字符结尾的连续⼦串** 。


**正确理解什么是前缀什么是后缀很重要** !


那么⽹上清⼀⾊都说 “kmp 最⻓公共前后缀” ⼜是什么回事呢？


我查了⼀遍 算法导论 和 算法4⾥KMP的章节，都没有提到 “最⻓公共前后缀”这个词，也不知道从哪⾥来了，我理

解是⽤“最⻓相等前后缀” 更准确⼀些。


**因为前缀表要求的就是相同前后缀的⻓度。**


⽽最⻓公共前后缀⾥⾯的“公共”，更像是说前缀和后缀公共的⻓度。这其实并不是前缀表所需要的。


所以字符串a的最⻓相等前后缀为0。

字符串aa的最⻓相等前后缀为1。

字符串aaa的最⻓相等前后缀为2。

等等.....。

class Solution {


public:


void getNext (int* next, const string& s){


next[0] = -1;


int j = -1;


for(int i = 1;i < s.size(); i++){


while(j >= 0 && s[i] != s[j + 1]) {


j = next[j];


}


if(s[i] == s[j + 1]) {


j++;


}


next[i] = j;


}


}


bool repeatedSubstringPattern (string s) {


if (s.size() == 0) {


return false;


}


int next[s.size()];


getNext(next, s);


int len = s.size();


if (next[len      - 1] != -1 && len % (len      - (next[len      - 1] + 1)) == 0) {


return true;


}


return false;


}


};
